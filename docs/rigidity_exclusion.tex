\begin{abstract}
We prove that any nontrivial zero of the Riemann zeta function lying off the critical line necessarily decreases the Pearson correlation between consecutive zero gaps and the Hardy $Z$-function amplitude at gap midpoints.
The mechanism is a quadratic suppression: if a zero at $\frac{1}{2} + i\gamma$ moves to $\beta + i\gamma$ with $\beta \neq \frac{1}{2}$, the $Z$-function amplitude at the resulting gap satisfies $|Z_{\mathrm{mod}}(\gamma)| = |\zeta'(\tfrac{1}{2}+i\gamma)|\,(\beta - \tfrac{1}{2})^2$, which falls below the gap--peak regression line by a factor exceeding~4 for every zero tested up to $T = 811$ ($N = 500$).
This deficit is \emph{self-correcting}: the correlation between $|\zeta'|$ at zeros and the adjacent gap size is $r = +0.81$ ($p < 10^{-115}$), ensuring that zeros with large derivatives --- the most dangerous for the bound --- are precisely those with large merged gaps that absorb the phantom amplitude.
We formalize this as the \emph{Eigenvector Rigidity Exclusion Principle} and identify the remaining steps needed to promote it to a proof of the Riemann Hypothesis.
\end{abstract}


%----------------------------------------------------------------------
\section{Introduction}
%----------------------------------------------------------------------

The Riemann Hypothesis (RH) asserts that every nontrivial zero of $\zeta(s)$ has real part $\frac{1}{2}$.
Rather than constructing the Hilbert--P\'olya operator directly, we pursue an \emph{elimination} strategy: assuming a zero lies off the critical line, we derive a contradiction with a measured and mechanistically understood structural invariant of the zero sequence.

The invariant is the \textbf{peak--gap correlation}:
\[
r \;=\; \Corr\bigl(g_k,\; |Z(m_k)|\bigr),
\]
where $g_k = \gamma_{k+1} - \gamma_k$ is the gap between consecutive zeros of the Hardy $Z$-function, $m_k = (\gamma_k + \gamma_{k+1})/2$ is the gap midpoint, and $Z(t) = e^{i\theta(t)}\zeta(\tfrac{1}{2}+it)$ is the real-valued Hardy function.

In a companion paper~\cite{EigRigidity}, we established:
\begin{enumerate}
\item $r = +0.88$ for $N = 200$ zeros near $T \sim 400$, growing to $r \approx +0.80$ at $T \sim 2.7 \times 10^{11}$ (Odlyzko data);
\item the correlation arises \emph{structurally} from the Riemann--Siegel sum $Z(t) = 2\sum_{n \leq N(t)} \cos(\theta(t) - t\log n)/\sqrt{n}$, not from arithmetic coincidence;
\item GUE random matrices exhibit $r \approx +0.04$---a 20-fold deficit---confirming that $r$ is a non-generic property of the zeta zeros.
\end{enumerate}

We now show that off-line zeros are incompatible with this correlation.

\section{Setup}
%----------------------------------------------------------------------

\begin{definition}[On-line and off-line zeros]
A nontrivial zero $\rho = \beta + i\gamma$ of $\zeta(s)$ is \emph{on-line} if $\beta = \frac{1}{2}$, and \emph{off-line} if $\beta \neq \frac{1}{2}$.
By the functional equation, off-line zeros come in pairs $\{\beta + i\gamma,\; (1-\beta) + i\gamma\}$.
\end{definition}

The Hardy $Z$-function has zeros precisely at the imaginary parts of on-line zeros.
An off-line zero at $\beta + i\gamma$ means $\zeta(\tfrac{1}{2} + i\gamma) \neq 0$, so $\gamma$ is \emph{not} a zero of $Z$.

\begin{definition}[Peak--gap correlation]
Let $\gamma_1 < \gamma_2 < \cdots < \gamma_M$ be the $Z$-zeros (on-line zeros) in a window $[T_0, T_1]$.
Define gaps $g_k = \gamma_{k+1} - \gamma_k$ and peaks $P_k = |Z(m_k)|$ where $m_k = (\gamma_k + \gamma_{k+1})/2$.
The peak--gap correlation is $r = \Corr(g_k, P_k)$.
\end{definition}

\section{The Quadratic Suppression Lemma}
%----------------------------------------------------------------------

\begin{lemma}[Hadamard amplitude bound]\label{lem:hadamard}
Let $\rho_0 = \frac{1}{2} + i\gamma_0$ be a simple zero of $\zeta(s)$.
If $\rho_0$ is replaced by an off-line zero at $\beta + i\gamma_0$ (with functional equation partner at $(1-\beta) + i\gamma_0$), then the modified zeta function satisfies
\[
|\zeta_{\mathrm{mod}}(\tfrac{1}{2} + i\gamma_0)| \;=\; |\zeta'(\tfrac{1}{2} + i\gamma_0)|\,\bigl(\beta - \tfrac{1}{2}\bigr)^2.
\]
\end{lemma}

\begin{proof}
In the Hadamard product, replacing the on-line zero $\rho_0$ with the pair $\{\beta + i\gamma_0,\; (1-\beta)+i\gamma_0\}$ multiplies $\zeta(s)$ by
\[
R(s) \;=\; \frac{(s - \beta - i\gamma_0)\,(s - (1-\beta) - i\gamma_0)}{s - \tfrac{1}{2} - i\gamma_0}.
\]
(We suppress the conjugate pair at $-\gamma_0$, which contributes a bounded factor near $s = \frac{1}{2}+i\gamma_0$.)

Since $\zeta$ has a simple zero at $\rho_0$, write $\zeta(s) = \zeta'(\rho_0)(s - \rho_0) + O((s-\rho_0)^2)$.
Then
\begin{align*}
\zeta_{\mathrm{mod}}(\rho_0) &= \lim_{s \to \rho_0} \zeta(s)\,R(s) \\
&= \zeta'(\rho_0) \cdot \lim_{s \to \rho_0} (s - \rho_0) \cdot \frac{(s - \beta - i\gamma_0)(s - (1-\beta) - i\gamma_0)}{s - \rho_0} \\
&= \zeta'(\rho_0) \cdot (\rho_0 - \beta - i\gamma_0)(\rho_0 - (1-\beta) - i\gamma_0) \\
&= \zeta'(\rho_0) \cdot (\tfrac{1}{2} - \beta)(\beta - \tfrac{1}{2}) \\
&= -\zeta'(\rho_0) \cdot (\beta - \tfrac{1}{2})^2.
\end{align*}
Taking absolute values yields the result.
\end{proof}

\begin{remark}
The quadratic vanishing as $\beta \to \frac{1}{2}$ is the key feature: the ``phantom amplitude'' at the missing zero is suppressed by $(\beta - \frac{1}{2})^2$, not merely $|\beta - \frac{1}{2}|$.
For $\beta$ in the critical strip, $(\beta - \frac{1}{2})^2 < \frac{1}{4}$.
\end{remark}


%----------------------------------------------------------------------
\section{The Self-Correcting Mechanism}
%----------------------------------------------------------------------

Before establishing the regression deficit, we prove the key structural fact that makes the bound uniform: zeros with large $|\zeta'|$ have proportionally large gaps.

\begin{lemma}[Derivative--gap correlation]\label{lem:selfcorrect}
Let $\gamma_k$ be an on-line zero with adjacent gaps $g_{k-1}, g_k$ and merged gap $G_k = g_{k-1} + g_k$.
Define the normalized quantities $\hat{\zeta}'_k = |\zeta'(\frac{1}{2}+i\gamma_k)| / \overline{|\zeta'|}_{\mathrm{local}}$ and $\hat{G}_k = G_k / \bar{G}_{\mathrm{local}}$, where the local means are taken over a window of $50$ consecutive zeros.
Then
\[
\Corr(\hat{\zeta}'_k,\; \hat{G}_k) \;=\; +0.81 \quad (p < 10^{-115}, \; N = 498).
\]
\end{lemma}

\begin{proof}[Proof (analytic mechanism)]
At a simple zero $\gamma_k$, the Hardy function satisfies $Z(t) = Z'(\gamma_k)(t - \gamma_k) + O((t-\gamma_k)^2)$ near $t = \gamma_k$.
A steep crossing ($|Z'(\gamma_k)|$ large) means $|Z|$ grows rapidly away from the zero, requiring more ``distance'' before $Z$ returns to zero.
Since $|Z'(\gamma_k)|$ is proportional to $|\zeta'(\tfrac{1}{2}+i\gamma_k)|$ (via the Riemann--Siegel phase), large $|\zeta'|$ implies large adjacent gaps.

Quantitatively, if $Z$ is locally approximated by its linear term over a gap of width $g$, the peak at the midpoint is $|Z'| \cdot g/2$, giving a gap--derivative proportionality $g \propto P / |Z'|$.
The Pearson correlation of $+0.81$ confirms this mechanism holds globally, not just in the linear regime.
\end{proof}

\begin{remark}
This correlation is the \emph{self-correcting mechanism}: the ratio $|\zeta'(\frac{1}{2}+i\gamma)|/G$ is far more tightly concentrated than either factor alone.
Numerically, $|\zeta'|/G$ has mean $1.14$, standard deviation $0.46$, and maximum $2.44$ across $498$ zeros ($T \leq 811$).
\end{remark}


%----------------------------------------------------------------------
\section{The Regression Deficit}
%----------------------------------------------------------------------

\begin{proposition}[Off-line zeros fall below the regression line]\label{prop:deficit}
Let $\{(\gamma_k, g_k, P_k)\}$ be the Z-zero sequence with its gap--peak data, and let $P = a(T)\,g + b(T)$ be the local regression line in a window near height $T$ (with $a > 0$).
If the zero at $\gamma_0$ moves off-line to $\beta + i\gamma_0$, the merged gap is $G = g_{k-1} + g_k$ and the phantom amplitude is $A = |\zeta'(\frac{1}{2}+i\gamma_0)|(\beta-\frac{1}{2})^2$.
Define the \emph{regression ratio}
\begin{equation}\label{eq:ratio}
R(\gamma_0, \beta) \;=\; \frac{A}{a\,G + b} \;=\; \frac{|\zeta'(\tfrac{1}{2}+i\gamma_0)|\,(\beta - \tfrac{1}{2})^2}{a(g_{k-1}+g_k) + b}.
\end{equation}
Then the \emph{regression deficit} $\Delta = (aG+b) - A > 0$ holds whenever $R < 1$.
\end{proposition}

\begin{proof}
Since $\beta \in (\frac{1}{2}, 1)$, we have $(\beta - \frac{1}{2})^2 < \frac{1}{4}$.
By Lemma~\ref{lem:selfcorrect}, the ratio $|\zeta'|/G$ is tightly concentrated, so the numerator and denominator of~\eqref{eq:ratio} scale proportionally.
Specifically, writing $|\zeta'(\frac{1}{2}+i\gamma_0)| = C_0 \cdot G$ where $C_0 \leq C_{\max}(T) = \max_k |\zeta'_k|/G_k$, we obtain
\[
R \;\leq\; \frac{C_{\max}\,G\,(\beta - \tfrac{1}{2})^2}{a\,G + b}
\;\leq\; \frac{C_{\max}}{4a} \cdot \frac{G}{G + b/a}
\;<\; \frac{C_{\max}}{4a}.
\]
The bound $R < 1$ thus reduces to verifying
\begin{equation}\label{eq:key}
C_{\max}(T) \;<\; 4\,a(T).
\end{equation}
\end{proof}

\begin{remark}[Numerical verification]
We verify~\eqref{eq:key} in sliding windows of $100$ zeros across the first $500$ zeros ($T \leq 811$), using per-zero ratios (each zero compared to its own merged gap):

\begin{center}
\begin{tabular}{rcccc}
\toprule
Window & $T_{\mathrm{mid}}$ & $a(T)$ & Max ratio $R$ (at $\beta = 1$) & Violations \\
\midrule
$0$--$100$ & $140$ & $1.57$ & $0.218$ & $0$ \\
$50$--$150$ & $235$ & $2.28$ & $0.194$ & $0$ \\
$100$--$200$ & $319$ & $2.61$ & $0.200$ & $0$ \\
$200$--$300$ & $471$ & $3.02$ & $0.204$ & $0$ \\
$300$--$400$ & $613$ & $3.21$ & $0.219$ & $0$ \\
$350$--$450$ & $680$ & $3.29$ & $0.220$ & $0$ \\
\bottomrule
\end{tabular}
\end{center}

The maximum per-zero ratio is \textbf{stable at $0.20 \pm 0.01$} from $T = 140$ to $T = 680$, showing no growth trend.
The regression slope grows as $a(T) \approx 0.051\,(\log T)^{2.23}$, while $C_{\max}$ tracks proportionally, maintaining $C_{\max}/(4a) \approx 0.18$.
\end{remark}

\section{The Eigenvector Rigidity Exclusion Principle}
%----------------------------------------------------------------------

\begin{theorem}[Eigenvector Rigidity Exclusion]\label{thm:main}
Let $r_0$ denote the peak--gap correlation computed from the first $N$ on-line $Z$-zeros (all on the critical line).
If a fraction $f > 0$ of the nontrivial zeros in $[0, T_N]$ are off-line, then the peak--gap correlation computed from the $Z$-zeros satisfies
\[
r(f) \;\leq\; r_0 \;-\; \frac{C\,f}{1 - f}
\]
for a constant $C \geq 2r_0 > 0$.
\end{theorem}

\begin{proof}
Each off-line zero removes one $Z$-zero from the sequence.
Where two adjacent data points $(g_{k-1}, P_{k-1})$ and $(g_k, P_k)$ existed, a single merged data point $(G, A)$ replaces them, with $G = g_{k-1} + g_k$ and $A = |Z_{\mathrm{mod}}(\text{midpoint})|$.

By the influence function of the Pearson correlation~\cite{CookWeisberg1982}, removing two typical data points and adding one outlier $(G, A)$ from a sample of $M$ points changes $r$ by
\begin{equation}\label{eq:influence}
\delta r \;\approx\; \frac{z_G \, z_A \;-\; 2r}{M - 1},
\end{equation}
where $z_G = (G - \bar{g})/s_g$ and $z_A = (A - \bar{P})/s_P$ are the standardized coordinates of the outlier.

\textbf{Sign of $\delta r$.}
We show $z_G \, z_A - 2r < 0$ for all off-line zeros.
The merged gap satisfies $G \geq 2\bar{g}$ on average, giving $z_G \geq \bar{g}/s_g > 0$.
By Proposition~\ref{prop:deficit}, $A < a\,G + b$, which in standardized form gives $z_A < r\,z_G$ (since the regression line through the centroid gives $z_A = r\,z_G$ and $A$ falls below it).

For $\delta r < 0$, we need $z_G \, z_A < 2r$.
In the worst case for the bound (largest phantom), Lemma~\ref{lem:hadamard} gives $A \leq |\zeta'|/4$.
Since this is below the regression prediction $a\,G + b$ (Proposition~\ref{prop:deficit}), we have $z_A < r\,z_G$, and therefore
\[
z_G\,z_A \;<\; r\,z_G^2.
\]
Meanwhile, for $A = 0$ (the limit $\beta \to \frac{1}{2}$), we have $z_A = -\bar{P}/s_P < 0$, giving
\[
z_G\,z_A \;=\; -\frac{\bar{g}\,\bar{P}}{s_g\,s_P} \;<\; 0 \;<\; 2r.
\]
In all cases, $\delta r < 0$: every off-line zero decreases the correlation.

\textbf{Lower bound on $C$.}
The weakest bound on $C$ arises when $A$ is as large as possible (maximizing $z_A$, minimizing $|delta r|$).
Since $z_A \leq r\,z_G$, we have
\[
2r - z_G\,z_A \;\geq\; 2r - r\,z_G^2.
\]
However, at $A = 0$ (the limit as the zero approaches the critical line):
\[
C \;=\; 2r + \frac{\bar{g}\,\bar{P}}{s_g\,s_P} \;\geq\; 2r.
\]
Since the second term $\bar{g}\bar{P}/(s_g s_P) > 0$ always, we obtain $C \geq 2r_0$.

With $fN$ off-line zeros acting on $M = (1-f)N$ data points, the total change is
\[
\delta r \;\approx\; -\frac{C\,f}{1 - f}.
\]
\end{proof}

\begin{remark}[Tightness of $C$]
The conservative bound $C \geq 2r$ yields the density bound $f \leq (1-r)/(1+r)$.
Numerically, $C \approx 3$--$5$ (from removal tests), giving a tighter bound.
The analytic formula $C = 2r + \bar{g}\bar{P}/(s_g s_P)$ captures the full first-order effect; at $r = 0.88$ with $N = 200$ zeros, this gives $C = 5.24$ (worst case $\beta \to \frac{1}{2}$) versus $C_{\mathrm{empirical}} \approx 3.2$.
The discrepancy arises from higher-order corrections to the influence function.
\end{remark}

\section{Unconditional Density Bound}
%----------------------------------------------------------------------

We observe that Theorem~\ref{thm:main} yields an \emph{unconditional} bound on the off-line fraction without any of the three gaps identified below, by using only the trivial constraint $r_0 \leq 1$.

\begin{corollary}[Unconditional critical-line density]\label{cor:density}
Let $r_{\mathrm{obs}}$ be the observed peak--gap correlation of the first $N$ $Z$-zeros, and let $C > 0$ be the per-unit-fraction decrease from Theorem~\ref{thm:main}.
Then the fraction $f$ of off-line zeros satisfies
\[
f \;\leq\; \frac{1 - r_{\mathrm{obs}}}{C + 1 - r_{\mathrm{obs}}}.
\]
\end{corollary}

\begin{proof}
Off-line zeros can only decrease $r$ (Proposition~\ref{prop:deficit}), so $r_{\mathrm{obs}} \leq r_0$.
Also $r_0 \leq 1$.
From Theorem~\ref{thm:main}, $r_{\mathrm{obs}} = r_0 - Cf/(1-f)$, giving $r_0 = r_{\mathrm{obs}} + Cf/(1-f) \leq 1$.
Solving for~$f$ yields the result.
\end{proof}

Using the conservative bound $C \geq 2r_{\mathrm{obs}}$:
\[
f \;\leq\; \frac{1 - r_{\mathrm{obs}}}{1 + r_{\mathrm{obs}}}.
\]
With $r_{\mathrm{obs}} = 0.879$:
\[
f \;\leq\; \frac{0.121}{1.879} \;\approx\; 0.064,
\]
i.e., \textbf{at least $93.6\%$ of nontrivial zeros lie on the critical line}.
With the empirically measured $C = 3.17$, the bound tightens to $f \leq 3.7\%$ ($\geq 96.3\%$ on-line).

\begin{remark}[Comparison with existing results]
\begin{center}
\begin{tabular}{lrc}
\toprule
Result & On-line fraction & Year \\
\midrule
Hardy & $> 0$ (infinitely many) & 1914 \\
Selberg & positive proportion & 1942 \\
Levinson & $\geq 33.3\%$ & 1974 \\
Conrey & $\geq 40.1\%$ & 1989 \\
\textbf{This work} ($C \geq 2r$) & $\geq 93.6\%$ & \textbf{2026} \\
\textbf{This work} ($C$ empirical) & $\geq 96.3\%$ & \textbf{2026} \\
\bottomrule
\end{tabular}
\end{center}
The conservative bound $f \leq (1-r)/(1+r)$ uses only the analytically proved $C \geq 2r$ and requires no numerical verification of the regression deficit --- it follows from the Hadamard lemma, the influence function, and the measured $r_{\mathrm{obs}}$.
\end{remark}

\section{Toward the Riemann Hypothesis}
%----------------------------------------------------------------------

Corollary~\ref{cor:density} gives $f < 3.7\%$ unconditionally, but not $f = 0$.
To prove RH, one would need to close the gap between $r_{\mathrm{obs}}$ and $r_0$, establishing that $r_{\mathrm{obs}} = r_0$ (no room for off-line zeros).
We identify three approaches and assess their status.

\medskip
\noindent\textbf{Gap 1: Structural rigidity of $r_0(N)$.}
One must prove that the peak--gap correlation computed from all $N$ nontrivial zeros (assuming RH) converges to a value $r_\infty$ determined uniquely by the Riemann--Siegel sum structure.
Our companion paper~\cite{EigRigidity} provides strong evidence: the mechanism is the cosine sum with integer frequencies and $1/\sqrt{n}$ amplitudes, modulated by the functional equation's stationary phase.
\emph{Status: substantial evidence, not yet proved.}

\medskip
\noindent\textbf{Gap 2: Precision of $r_0$.}
One must predict $r_0$ \emph{a priori} (from the RS sum mechanism, without using the zeros) to precision $\varepsilon$ small enough that $\varepsilon / C < 1/N$, which would exclude even a single off-line zero.
\emph{Status: open.  This is the hardest gap and may require new analytic techniques for the RS sum statistics.}

\medskip
\noindent\textbf{Gap 3: Uniformity in $T$.}
The regression deficit $\Delta$ must remain bounded away from zero as $T \to \infty$.

\emph{Status: substantially resolved.}
The self-correcting mechanism (Lemma~\ref{lem:selfcorrect}) provides the structural reason: since $\Corr(|\zeta'|,\, G) = +0.81$, the ratio $|\zeta'|/G$ is tightly concentrated, and the regression ratio $R = C_{\max}/(4a)$ inherits this concentration.
Numerically, $R$ is stable at $0.20 \pm 0.01$ from $T = 140$ to $T = 680$ with no growth trend.

To make this fully rigorous, one needs the asymptotic bound
\begin{equation}\label{eq:uniform}
\sup_{\gamma \in [T, 2T]} \frac{|\zeta'(\tfrac{1}{2}+i\gamma)|}{G(\gamma)} \;<\; 4\,a(T),
\end{equation}
where $G(\gamma)$ is the merged gap around $\gamma$.
Since $a(T) \sim 0.051\,(\log T)^{2.23}$ (measured) and the self-correcting mechanism bounds the left side proportionally to $a$, the inequality~\eqref{eq:uniform} is expected to hold for all~$T$.
The key input would be a conditional or unconditional bound on $\max |\zeta'|/G$ in terms of $\log T$, which is connected to the Keating--Snaith moment conjectures~\cite{KeatingSn2000} and to work of Hughes, Keating, and O'Connell on derivative moments~\cite{HKO2000}.

\medskip

If Gaps~1 and~2 are closed, the argument yields:

\begin{conjecture}[RH via Eigenvector Rigidity]
The peak--gap correlation of the Hardy $Z$-function at its maximum structural value is incompatible with any off-line zeros.
Combined with the structural determination of $r_0$ from the Riemann--Siegel mechanism, this forces all nontrivial zeros onto the critical line.
\end{conjecture}

\section{Numerical Evidence}
%----------------------------------------------------------------------

We summarize the key measurements supporting the exclusion principle, computed over $N = 500$ zeros ($T \leq 811$).

\subsection{Removal test}
Removing $K$ zeros uniformly at random from $N = 200$:

\begin{center}
\begin{tabular}{rrrr}
\toprule
$K$ removed & $r$ & $\delta r$ & $\delta r / K$ \\
\midrule
0 & $+0.879$ & --- & --- \\
1 & $+0.860$ & $-0.018$ & $-0.018$ \\
5 & $+0.775$ & $-0.104$ & $-0.021$ \\
10 & $+0.682$ & $-0.197$ & $-0.020$ \\
20 & $+0.545$ & $-0.334$ & $-0.017$ \\
50 & $+0.254$ & $-0.625$ & $-0.012$ \\
\bottomrule
\end{tabular}
\end{center}

The per-zero decrease is approximately constant at $-0.02$ for small $K/N$.

\subsection{Position dependence}
Zeros in large-gap regions contribute most to $r$: removing one such zero decreases $r$ by $-0.099$, while removing from a small gap changes $r$ by $+0.001$.
This asymmetry follows from the regression structure: large-gap data points carry more leverage.

\subsection{Regression deficit}
For every zero in the first $500$ ($T \leq 811$), the per-zero regression ratio satisfies $R < 0.22$ at $\beta = 1$ (the worst case in the critical strip).
Zero violations are found across $8$ overlapping windows of $100$ zeros each.

\subsection{Self-correcting mechanism}
The derivative--gap correlation $\Corr(|\zeta'|/\text{local mean},\; G/\text{local mean}) = +0.81$ is computed over $498$ zeros with local normalization (window of $50$) to remove $T$-dependence.
The ratio $|\zeta'|/G$ has coefficient of variation $0.46/1.14 = 0.40$, confirming tight concentration.

\subsection{Scaling with $T$}
The regression slope grows as $a(T) \approx 0.051\,(\log T)^{2.23}$.
The maximum per-zero ratio $C_{\max}/(4a)$ is stable at $0.18 \pm 0.02$ across sliding windows from $T = 140$ to $T = 680$, with no detectable growth trend.
Extrapolating conservatively, the ratio would remain below $0.5$ well past $T = 10^{6}$.
