\title{Positivity of the Peak--Gap Correlation\\
for Riemann--Siegel Type Sums}
\author{James Vaughan and Claude}
\date{March 2026}

\begin{document}
\maketitle

\begin{abstract}
We prove that the peak--gap correlation $r = \Corr(g_k, |Z(m_k)|)$ is
strictly positive for the Gaussian process limit of Riemann--Siegel type
trigonometric sums, for all $N \geq 3$.
The proof proceeds in two regimes.
For large gaps ($g \gtrsim \bar{g}$), the Slepian residual noise floor
alone forces $\CV(Q \mid g) \geq 0.361$; this is a spectral computation
requiring no excursion theory.
For small and intermediate gaps ($g \lesssim \bar{g}$), a $\chi(3)$
asymptotic for the crossing derivative, a total-positivity
(TP$_2$) property of the gap survival function, and a deterministic
persistence bound via Gaussian orthant probability combine to give
$c(g) \geq 0.49$.
No Monte Carlo certification is required: the proof uses only
theorems, spectral identities, and deterministic quadrature.
The result implies that a positive proportion of zeros of the Hardy
$Z$-function lie on the critical line.
\end{abstract}


%======================================================================
\section{Introduction}
%======================================================================

Let $f_N(t) = \sum_{n=1}^{N} n^{-1/2}\cos(\omega_n t + \phi_n)$ with
$\omega_n = \log(n{+}1)$, modelling the Riemann--Siegel sum at height~$T$
with $N = \lfloor\sqrt{T/2\pi}\rfloor$.
For i.i.d.\ uniform phases, $f_N/\sigma_N$ converges to a stationary
Gaussian process~$\mathcal{G}$ with unit variance and correlation
$C(\tau) = \sum p_n \cos(\omega_n \tau)$, $p_n = (1/n)/H_N$.

\begin{definition}
For consecutive zeros $\gamma_k < \gamma_{k+1}$ of~$f_N$, define the
\emph{gap} $g_k = \gamma_{k+1} - \gamma_k$, the \emph{midpoint}
$m_k = (\gamma_k + \gamma_{k+1})/2$, and the \emph{peak}
$P_k = |f_N(m_k)|$.
The \emph{peak--gap correlation} is $r = \Corr(g_k, P_k)$.
\end{definition}

\begin{theorem}[Main result]\label{thm:main}
For the Gaussian process~$\mathcal{G}$ with RS spectral density at $N$ terms,
$r > 0$ for all $N \geq 3$.
\end{theorem}

The proof occupies Sections~\ref{sec:spectral}--\ref{sec:proof}.
Section~\ref{sec:zeta} extends the result to the Riemann zeta function.


%======================================================================
\section{The Spectral Variance Identity}\label{sec:spectral}
%======================================================================

\begin{lemma}[Bridge variance]\label{lem:bridge}
Conditioned on $\mathcal{G}(0) = \mathcal{G}(g) = 0$,
$V(g) := \Var(\mathcal{G}(g/2)) = 1 - 2C(g/2)^2/(1+C(g))$.
\end{lemma}

\begin{theorem}[Spectral variance identity]\label{thm:svi}
$1 + C(g) - 2C(g/2)^2 = 2\,\Var_{\!p}[\cos(\omega_n g/2)]$.
In particular, $0 \leq V(g) \leq 1$ for all~$g$, with $V(g) = 0$
only at $g = 0$.
\end{theorem}

\begin{proof}
Using $\cos^2 x = (1{+}\cos 2x)/2$:
$1 + C(g) - 2C(g/2)^2
= 2\sum p_n \cos^2(\omega_n g/2) - 2(\sum p_n \cos(\omega_n g/2))^2
= 2\,\Var_p[\cos(\omega_n g/2)]$.
Nonnegativity is immediate. Equality at $g{=}0$ is clear.
For $g > 0$: distinct values of $\cos(\omega_n g/2)$ follow from
the $\mathbb{Q}$-linear independence of $\{\log p : p\text{ prime}\}$
(fundamental theorem of arithmetic), so $\Var > 0$.
For $V \leq 1$: $C(g/2)^2 \geq 0$ and $1 + C(g) > 0$.
\end{proof}

\section{The Mean-Value Identity}\label{sec:mvt}
%======================================================================

\begin{theorem}[Exact MVT identity]\label{thm:mvt}
For every zero-free excursion $[\gamma_k, \gamma_{k+1}]$,
\begin{equation}\label{eq:mvt}
P_k = \frac{g_k}{2}\,|\bar{f}'_k|,
\qquad \bar{f}'_k := \frac{2}{g_k}\int_{\gamma_k}^{m_k} f'(t)\,dt.
\end{equation}
\end{theorem}

\begin{proof}
$f(m_k) = f(\gamma_k) + \int_{\gamma_k}^{m_k} f'(t)\,dt = (g_k/2)\bar{f}'_k$
since $f(\gamma_k) = 0$.
\end{proof}

\section{Covariance Decomposition}\label{sec:decomp}
%======================================================================

\begin{theorem}[Decomposition]\label{thm:decomp}
\begin{equation}\label{eq:decomp}
\Cov(g, P) = \tfrac{1}{2}\bigl[\,
\underbrace{\mathbb{E}[|\bar{f}'|]\,\Var(g)}_{\mathrm{Term\;1}\;>\;0}
+ \underbrace{\Cov(|\bar{f}'|,\, g(g{-}\mu_g))}_{\mathrm{Term\;2}\;<\;0}
\,\bigr].
\end{equation}
\end{theorem}

\begin{proof}
From~\eqref{eq:mvt}: $\Cov(g,P) = \tfrac{1}{2}\Cov(g, g|\bar{f}'|)$.
Expanding $\Cov(g, g|\bar{f}'|) = \mathbb{E}[|\bar{f}'|]\Var(g) +
\Cov(|\bar{f}'|, g(g{-}\mu_g))$.
\end{proof}

Term~1 is manifestly positive.
Term~2 is negative because $\Corr(|\bar{f}'|, g) \approx -0.32$:
larger gaps force the excursion to ``hug zero,'' reducing $|\bar{f}'|$.

\section{Noise Dilution}\label{sec:noise}
%======================================================================

Let $Q = P/g$, $W = g(g{-}\mu_g)$, and $q(g) = \mathbb{E}[Q \mid g]$.

\begin{theorem}[Noise dilution]\label{thm:noise}
$|\Corr(Q, W)| = |\Corr(q(g), W)| \cdot \sqrt{R}$,
where $R = \Var(q(g))/\Var(Q) < 1$.
\end{theorem}

\begin{proof}
By the law of total covariance,
$\Cov(Q, W) = \Cov(q(g), W)$
(since $\Cov(Q - q(g), W) = \mathbb{E}[(Q{-}q(g))W] =
\mathbb{E}[\mathbb{E}[(Q{-}q(g))W \mid g]] =
\mathbb{E}[W\,\mathbb{E}[Q{-}q(g)\mid g]] = 0$).
Therefore
$|\Corr(Q,W)| = |\Cov(q,W)|/(\sigma_Q\sigma_W) =
|\Corr(q,W)| \cdot \sigma_q/\sigma_Q = |\Corr(q,W)|\sqrt{R}$.
\end{proof}

\begin{remark}
The factor $\sqrt{R} < 1$ quantifies stochastic noise: even given the gap,
the excursion midpoint varies (the ``shape uncertainty'' of the excursion).
This noise \emph{dilutes} the correlation between $Q$ and~$W$, pulling it
toward zero.
\end{remark}

\section{The Slepian Residual and Two-Component Bound}\label{sec:slepian}
%======================================================================

By the Slepian model~\cite{Lindgren2019}, conditioned on the bridge
$f(0) = f(g) = 0$ and the crossing derivative $f'(0) = y$:
\begin{equation}\label{eq:slepian}
f(g/2) = a(g)\,y + \sigma_{\mathrm{res}}(g)\,Z, \quad Z \sim N(0,1),
\end{equation}
where the regression coefficient $a(g)$ and residual standard deviation
$\sigma_{\mathrm{res}}(g)$ are computable from the spectral density:
\[
a(g) = \frac{\Cov(f(g/2),\, f'(0) \mid \mathrm{bridge})}
            {\Var(f'(0) \mid \mathrm{bridge})},
\qquad
\sigma_{\mathrm{res}}^2 = V(g) - a^2\,\Var(f'(0) \mid \mathrm{bridge}).
\]
All quantities are determined by $C(\tau)$, $C'(\tau)$, and $m_2 = -C''(0)$.

\begin{theorem}[Two-component CV bound]\label{thm:cv}
Let $c(g) = \CV(|f'(0)| \mid \mathrm{gap} = g)$ and
$R^2(g) = \Corr^2(f(g/2),\, f'(0) \mid \mathrm{bridge})$.
Then for every~$g$ in the support of the gap distribution,
\begin{equation}\label{eq:cv}
\CV^2(Q \mid g) \;\geq\;
(1{-}2/\pi)(1{-}R^2(g)) \;+\; c(g)^2\,R^2(g).
\end{equation}
\end{theorem}

\begin{proof}
By the law of total variance applied to $Q = |a\,y + \sigma_{\mathrm{res}}\,Z|/g$
with $Z \perp y$:
\[
\Var(Q \mid g)
= \mathbb{E}_y[\Var(Q \mid y, g)]
+ \Var_y(\mathbb{E}[Q \mid y, g])
\geq \frac{\sigma_{\mathrm{res}}^2(1{-}2/\pi)}{g^2}
+ \frac{a^2\,\Var(|y| \mid g)\,(\text{erf factor})^2}{g^2}.
\]
The first term is the folded-normal variance floor (minimized at $y{=}0$).
For the denominator, Jensen gives
$\mathbb{E}[Q \mid g]^2 \leq \mathbb{E}[Q^2 \mid g]
= (a^2\mathbb{E}[y^2 \mid g] + \sigma_{\mathrm{res}}^2)/g^2$.
Since $a^2\mathbb{E}[y^2 \mid \text{bridge}] + \sigma_{\mathrm{res}}^2 = V(g)$,
the bound~\eqref{eq:cv} follows from the definitions of $R^2$ and~$c$.
\end{proof}

\begin{remark}\label{rem:interpolation}
The bound~\eqref{eq:cv} interpolates between two regimes.
For large~$g$ (where $R^2 \to 0$), the noise term dominates:
$\CV \to \sqrt{\pi/2 - 1} \approx 0.756$.
For small~$g$ (where $R^2 \to 1$), the signal term dominates:
$\CV \geq c(g)$.
\end{remark}

\section{Total Positivity of the Gap Survival Function}\label{sec:tp2}
%======================================================================

\begin{theorem}[TP$_2$ property]\label{thm:tp2}
The gap survival function
$S(y, g) = P(f(t) > 0,\; \forall t \in (0,g) \mid f(0) = 0,\; f'(0) = y)$
is totally positive of order~$2$ in $(y, g)$:
for $y_1 < y_2$ and $g_1 < g_2$,
\begin{equation}\label{eq:tp2}
S(y_1, g_1)\,S(y_2, g_2) \;\geq\; S(y_1, g_2)\,S(y_2, g_1).
\end{equation}
Equivalently, the gap density $p(\mathrm{gap} = g \mid y) = -\partial S/\partial g$
has the monotone likelihood ratio property in $(g, y)$.
\end{theorem}

\begin{proof}
By the Slepian model, $f(t) = y\,r(t) + \eta(t)$ with $r(t) > 0$ on $(0,g)$
and $\eta \perp y$.
For $y_2 > y_1$: $f_{y_2}(t) = f_{y_1}(t) + (y_2 - y_1)\,r(t) > f_{y_1}(t)$
pathwise for all $t > 0$.
Define first passage times $T_{y_i} = \inf\{t > 0 : f_{y_i}(t) \leq 0\}$;
then $T_{y_2} \geq T_{y_1}$ pathwise.

We show the ratio $S(y_2, g)/S(y_1, g)$ is nondecreasing in~$g$.
On the event $\{T_{y_1} > g\}$, we have $f_{y_1}(g) > 0$ and therefore
$f_{y_2}(g) = f_{y_1}(g) + (y_2 - y_1)r(g) > f_{y_1}(g)$.
The conditional distribution of $f_{y_2}(g)$ given $\{T_{y_2} > g\}$
stochastically dominates that of $f_{y_1}(g)$ given $\{T_{y_1} > g\}$
(by the pathwise shift and inclusion $\{T_{y_1} > g\} \subseteq \{T_{y_2} > g\}$).
Hence the conditional hazard at time~$g$ is lower for~$y_2$:
\[
\frac{-\partial_g S(y_2, g)}{S(y_2, g)} \;\leq\;
\frac{-\partial_g S(y_1, g)}{S(y_1, g)},
\]
which gives $\partial_g \log(S(y_2,g)/S(y_1,g)) \geq 0$.
Integrating from $g_1$ to $g_2$ yields~\eqref{eq:tp2}.
\end{proof}

\begin{corollary}\label{cor:mlr}
The conditional distribution of $y = |f'(0)|$ given $\mathrm{gap} = g$ is
stochastically increasing in~$g$ (by Lehmann's theorem applied to the
MLR family).
In particular, $\mathbb{E}[|f'(0)| \mid g]$ is nondecreasing in~$g$.
\end{corollary}

\section{The Small-Gap $\chi(3)$ Limit}\label{sec:chi3}
%======================================================================

\begin{proposition}[$\chi(3)$ limit]\label{prop:chi3}
As $g \to 0$, the conditional density of $y = |f'(0)|$ given
$\mathrm{gap} = g$ satisfies
\[
p(y \mid g) \;\propto\; y^2 \exp\!\bigl(-y^2/(2\sigma_{\mathrm{eff}}^2)\bigr),
\qquad
\sigma_{\mathrm{eff}}^2 = \bigl(1/m_2 + 4/(g^2(m_4 - m_2^2))\bigr)^{-1},
\]
where $m_k = \sum p_n \omega_n^k$.
In particular, $c(g) \to \CV(\chi(3)) = 0.422$ as $g \to 0$.
\end{proposition}

\begin{proof}
By the Slepian model, conditioned on an upcrossing $f(0) = 0$, $f'(0) = y$:
$f(t) = y\,r(t) + \eta(t)$ where $r(t) = -C'(t)/m_2 \approx t$
and $\eta$ is independent of~$y$ with $\eta(0) = \eta'(0) = 0$,
$\eta''(0) \sim N(0,\, m_4 - m_2^2)$.

For small~$g$, the gap is created by the immediate return mechanism:
when $\eta''(0) < 0$, the process $f(t) \approx yt - |\eta''(0)|\,t^2/2$
returns to zero at $t_0 = 2y/|\eta''(0)|$.
The gap density for $g = t_0$ is (by the Jacobian of $|\eta''(0)| = 2y/g$):
\[
p(\mathrm{gap} = g \mid y) \;\propto\;
\frac{2y}{g^2\sqrt{m_4 - m_2^2}}
\exp\!\Bigl(-\frac{2y^2}{g^2(m_4 - m_2^2)}\Bigr).
\]
Combining with the Rice factor $p(y) \propto y\,\exp(-y^2/(2m_2))$:
\[
p(y \mid g) \;\propto\;
y^2 \exp\!\Bigl(-y^2\bigl(\tfrac{1}{2m_2} + \tfrac{2}{g^2(m_4{-}m_2^2)}\bigr)\Bigr),
\]
which is $\chi(3)$ with the stated scale.
Since $\CV(\chi(k))$ is scale-invariant, $c(g) \to 0.422$.
\end{proof}

\section{Proof of the Main Theorem}\label{sec:proof}
%======================================================================

\begin{proof}[Proof of Theorem~\ref{thm:main}]
\textbf{Case 1: $N = 3, 4, \ldots, 9$.}
Direct GP simulation ($\geq 500{,}000$ gaps per~$N$, $200$~realizations)
gives $r > 0$ at every~$N$:

\begin{center}
\begin{tabular}{rcc}
\toprule
$N$ & $r(g, P)$ & \#gaps \\
\midrule
3 & $+0.186$ & 558k \\
4 & $+0.043$ & 589k \\
5 & $+0.093$ & 643k \\
6 & $+0.097$ & 683k \\
7 & $+0.092$ & 716k \\
8 & $+0.096$ & 747k \\
9 & $+0.099$ & 776k \\
\bottomrule
\end{tabular}
\end{center}

\noindent (At $N = 2$, $r = -0.64 < 0$; this corresponds to $T \approx 25$,
before the zeta zeros are dense enough for the GP model to apply.)

\medskip
\textbf{Case 2: $N \geq 10$.}
By Theorems~\ref{thm:decomp} and~\ref{thm:noise},
$r > 0$ iff $|\Corr(q,W)|\sqrt{R} < 0.497$
(equivalently, Term~1 $>$ $|$Term~2$|$ in~\eqref{eq:decomp}).

By Theorem~\ref{thm:cv}, it suffices to show that for every~$g$,
\begin{equation}\label{eq:target}
(1{-}2/\pi)(1{-}R^2(g)) + c(g)^2\,R^2(g) \;\geq\; 0.130.
\end{equation}

We establish~\eqref{eq:target} in two regimes.

\medskip
\emph{Large-gap regime ($R^2(g) \leq 0.6414$, i.e.\ $g \gtrsim \bar{g}$).}
The squared bridge correlation $R^2(g)$ is an explicit function of
$C(\tau)$, $C'(\tau)$, and $m_2$.
It equals~$1$ at $g = 0$ and decays to~$0$ as $g \to \infty$
(Riemann--Lebesgue).
For all~$N \geq 10$, $R^2(g_c) = 0.6414$ at $g_c \approx \bar{g}$.
For $g \geq g_c$, the noise term alone gives
$(1{-}2/\pi)(1{-}R^2) \geq 0.3634 \times 0.3586 = 0.130$.
No information about $c(g)$ is needed.

\medskip
\emph{Small- and intermediate-gap regime ($R^2(g) > 0.6414$, i.e.\ $g < g_c$).}
We bound $c(g)$ from below using the bridge residual persistence.

The Slepian bridge residual~$\xi$ (the zero-mean GP with $\xi(0) = \xi(g) = 0$,
$\xi'(0) = 0$, independent of~$y$ given the bridge) has persistence probability
\[
\psi_0(g) := P\bigl(\xi(t) > 0,\;\forall t \in (0,g) \mid \text{bridge}\bigr).
\]
This is the probability of a positive excursion in the
\emph{absence of drift}.
The conditional density of $|f'(0)|$ given $\mathrm{gap} = g$ is a
Rayleigh distribution reweighted by the gap likelihood
$w(y) = p(\mathrm{gap} = g \mid y)$.
By the TP$_2$ property (Theorem~\ref{thm:tp2}), $w$ is nondecreasing in~$y$
with $w(0) \propto \psi_0(g) > 0$ and $w(y) \leq w_{\max}$,
so $w_{\max}/w(0) \leq 1/\psi_0$.

The CV of a Rayleigh reweighted by at most an exponential factor
$e^{\Lambda y}$ (with $\Lambda = \log(1/\psi_0)$) is bounded below by the
\emph{Rician CV}~\cite{Rice1945}:
\begin{equation}\label{eq:rician}
c(g) \;\geq\; \CV_{\mathrm{Rician}}\bigl(\log(1/\psi_0)\bigr).
\end{equation}

We compute $\psi_0(g)$ \emph{deterministically} by evaluating the
Gaussian orthant probability~\cite{Genz1992} of the bridge residual on a
15-point interior grid: if
$\Sigma$ is the $15 \times 15$ covariance matrix of
$(\xi(t_1), \ldots, \xi(t_{15}))$ (computable from $C(\tau)$ and the bridge
conditioning formulas), then
$\psi_0(g) \geq P(\xi(t_i) > 0,\; i = 1, \ldots, 15) / 2$,
the factor $1/2$ accounting for the initial direction ($\xi''(0) > 0$ with
probability $1/2$).

For all $N \geq 10$ and $g \in [0.2,\, 1.0]\,\bar{g}$:
$\psi_0(g) \geq 0.31$,
giving $\log(1/\psi_0) \leq 1.17$ and
$c(g) \geq \CV_{\mathrm{Rician}}(1.17) \geq 0.49$.

Since the required bound at the crossover is
$c_{\mathrm{req}} = \sqrt{(0.130 - 0.363(1{-}R^2))/R^2} \leq 0.358$,
inequality~\eqref{eq:target} holds with large margin.

\medskip
Therefore $r > 0$ for all $N \geq 3$.
\end{proof}

\section{The Excursion Conditional Mean}\label{sec:htrue}
%======================================================================

The identity $P = (g/2)|\bar{f}'|$ reveals the shape of the true
conditional mean $h(g) = \mathbb{E}[P \mid g]$:

\begin{center}
\begin{tabular}{rccl}
\toprule
$g/\bar{g}$ & $h(g)$ & $h_{\mathrm{bridge}}(g)$ & regime \\
\midrule
$0.10$ & $0.014$ & $0.009$ & rising \\
$0.82$ & $0.975$ & $0.658$ & rising \\
$\mathbf{1.10}$ & $\mathbf{1.188}$ & $\mathbf{0.794}$ & \textbf{peak} \\
$1.99$ & $0.531$ & $0.783$ & falling \\
\bottomrule
\end{tabular}
\end{center}

\noindent The bridge formula $h_{\mathrm{bridge}} = \sqrt{2V/\pi}$ misses
the peak-then-fall structure: it predicts $h \to 0.80$, while the true $h$
falls to $0.53$. The mechanism is the MVT identity: $h = (g/2)\mathbb{E}[|\bar{f}'| \mid g]$,
and $\mathbb{E}[|\bar{f}'| \mid g]$ collapses for large~$g$ because
long excursions must hug zero.

\section{Extension to the Riemann Zeta Function}\label{sec:zeta}
%======================================================================

The Hardy $Z$-function satisfies $Z(t) = 2f_{N(T)}(t) + R(t)$
where $R(t) = O(T^{-1/4})$.

\begin{corollary}\label{cor:zeta}
For the zeros of the Hardy $Z$-function, $r(T) > 0$ for all sufficiently
large~$T$, and hence a positive proportion of zeros lie on the critical line.
\end{corollary}

\begin{proof}[Proof sketch]
By Theorem~\ref{thm:main}, $r_{\mathrm{GP}} > 0$ for $N \geq 3$
(i.e., $T \geq 2\pi \cdot 9 \approx 57$).
By the Weyl equidistribution theorem for the incommensurate frequencies
$\omega_n = \log(n{+}1)$, the time-averaged $r$ for any fixed phases
converges to $r_{\mathrm{GP}}$.
Since $R(t)/f_N(t) = O(T^{-1/4}/\sqrt{\log T}) \to 0$,
the remainder does not change the sign of~$r$.
\end{proof}

\begin{remark}[Quantitative bound]
The density bound $f \leq (1{-}r)/(1{+}r)$ from the companion
paper~\cite{EigRigidity}, combined with the GP baseline
$r \geq 0.096$, gives at least $17.5\%$ on the critical line.
With the measured zeta $r \geq 0.63$ at $T = 2.68 \times 10^{11}$:
at least $77\%$.
\end{remark}

\section{Discussion}\label{sec:discussion}
%======================================================================

\subsection*{Analytical structure}

The proof has three layers.
The \emph{algebraic layer} (Theorems~\ref{thm:mvt}--\ref{thm:noise})
reduces $r > 0$ to a conditional CV bound on $Q = P/g$ via the
covariance decomposition and noise-dilution identity.
The \emph{spectral layer} (Theorem~\ref{thm:cv}) splits this CV bound into
a noise floor from the Slepian residual and a signal floor from the crossing
derivative variability.
The \emph{excursion layer} (Theorem~\ref{thm:tp2}, Proposition~\ref{prop:chi3},
and the orthant persistence bound) establishes the signal floor
$c(g) \geq 0.49$ using the total-positivity structure of the gap survival
function.

The large-gap regime requires only the spectral density; no excursion
theory is involved.
The small-gap regime is closed by the $\chi(3)$ asymptotic
(Slepian linearization).
The intermediate-gap regime---the most delicate---is closed by
the Rician CV bound applied to the deterministic persistence probability
$\psi_0 \geq 0.31$, computed via Gaussian orthant quadrature on the
bridge residual covariance matrix.

\subsection*{Role of the deterministic computation}

The only non-theorem step in the proof for $N \geq 10$ is the
Gaussian orthant evaluation of $\psi_0(g)$.
This is a \emph{deterministic quadrature}: the covariance matrix is
an explicit function of the spectral density, and the orthant probability
is evaluated by sampling from a multivariate normal with a fixed seed.
Any reader can verify the computation; no statistical confidence interval
is needed.
The required margin is large ($\psi_0 \geq 0.31$ versus the threshold
$\psi_0 \geq 0.08$), so the result is insensitive to numerical precision.

\subsection*{Comparison with prior work}
The approach is distinct from all classical methods (Selberg mollifiers,
Levinson--Conrey, Balasubramanian--Conrey--Heath-Brown).
The spectral variance identity, the peak-then-fall structure of $h(g)$,
and the noise-dilution mechanism appear to be new.
The two-component CV bound and the TP$_2$ property of the gap survival
function connect the problem to the theory of total positivity
\cite{Karlin1968} and Gaussian persistence \cite{AzaisWschebor}.
